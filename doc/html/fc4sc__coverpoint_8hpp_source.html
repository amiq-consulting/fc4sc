128  void sample(const T &cvp_val) {
129 #ifdef FC4SC_DISABLE_SAMPLING
130  return;
131 #endif
132  if (!collect) return;
133  this->last_sample_success = false;
134 
135  // 1) Search if the value is in the ignore bins
136  for (auto& ig_bin_it : ignore_bins) {
137  if (ig_bin_it.sample(cvp_val)) {
138  misses++;
139  return;
140  }
141  }
142  // 2) Search if the value is in the illegal bins
143  for (auto& il_bin_hit : illegal_bins) {
144  try { il_bin_hit.sample(cvp_val); }
145  catch (illegal_bin_sample_exception &e) {
146  e.update_cvp_info(this->name);
147  throw e;
148  }
149  }
150  // Sample regular bins
151  for (size_t i = 0; i < bins.size(); ++i) {
152  if (bins[i].sample(cvp_val)) {
153  this->last_bin_index_hit = i;
154  this->last_sample_success = true;
155  if (this->stop_sample_on_first_bin_hit) return;
156  }
157  }
158 
159  if (!this->last_sample_success) { misses++; }
160  }
